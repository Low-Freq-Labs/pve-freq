# This file describes the network interfaces available on your system
# and how to activate them. For more information, see interfaces(5).

source /etc/network/interfaces.d/*

# The loopback network interface
auto lo
iface lo inet loopback

# Service NIC — VLAN 5 (Public)
allow-hotplug ens18
iface ens18 inet static
	address 10.25.5.31/24
	gateway 10.25.5.1
	dns-nameservers 1.1.1.1 8.8.8.8
	mtu 9000
	# Static route: LAN + NFS traffic via switch SVI (bypass pfSense)
	up ip route add 10.25.0.0/24 via 10.25.5.5

# Management NIC — VLAN 2550
allow-hotplug ens19
iface ens19 inet static
	address 10.25.255.31/24
	mtu 9000

    up ip route add 10.25.100.0/24 via 10.25.255.1
